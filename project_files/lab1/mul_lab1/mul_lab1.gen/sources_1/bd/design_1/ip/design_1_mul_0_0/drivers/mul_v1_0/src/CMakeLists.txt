cmake_minimum_required(VERSION 3.15)
project(mul)
find_package(common)
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_SOURCES xmul.c)
collect (PROJECT_LIB_SOURCES xmul_sinit.c)
collect (PROJECT_LIB_SOURCES xmul_linux.c)
collect (PROJECT_LIB_SOURCES xmul_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_HEADERS xmul.h)
collect (PROJECT_LIB_HEADERS xmul_hw.h)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
include_directories(${CMAKE_BINARY_DIR}/include)
add_library(mul STATIC ${_sources})
set_target_properties(mul PROPERTIES LINKER_LANGUAGE C)
